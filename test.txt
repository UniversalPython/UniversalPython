#!/bin/bash

# 1. Define your test files (Update these paths to match your local setup)
NATIVE_FILE="/test/samples/de/hello.de.py"
ENGLISH_FILE="/test/samples/en/conditionals.py"
LANGUAGE="german"

echo "--- Phase 1: Native to English Transpilation ---"
# Run the command. Replace 'upython' with the actual command used in the repo
# (e.g., 'python3 main.py' or 'universalpython')
universalpython --lang $LANGUAGE $NATIVE_FILE > /dev/null 2>&1

if [ $? -eq 0 ]; then
    echo "SUCCESS: Transpilation did not crash."
else
    echo "FAILED: Transpilation crashed!"
    exit 1
fi

echo ""
echo "--- Phase 2: English to Native (Reverse) ---"
# Run the reverse command
universalpython --reverse --lang $LANGUAGE $ENGLISH_FILE > /dev/null 2>&1

if [ $? -eq 0 ]; then
    echo "SUCCESS: Reverse transpilation did not crash."
else
    echo "FAILED: Reverse transpilation crashed!"
    exit 1
fi

echo ""
echo "Test complete. No crashes detected."
