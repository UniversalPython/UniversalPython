#!/bin/bash

NATIVE_FILE="/test/samples/fr/hello.fr.py"
ENGLISH_FILE="/test/samples/en/conditionals.py"
LANGUAGE="french"

echo "Native to English Transpilation"
universalpython -sl $LANGUAGE $NATIVE_FILE > /dev/null 2>&1

if [ $? -eq 0 ]; then
    echo "Transpilation did not crash"
else
    echo "Transpilation crashed"
    exit 1
fi

echo ""
echo "English to Native"
universalpython -r -sl $LANGUAGE $ENGLISH_FILE > /dev/null 2>&1

if [ $? -eq 0 ]; then
    echo "Reverse transpilation did not crash"
else
    echo "Reverse transpilation crashed!"
    exit 1
fi

echo ""
echo "Test complete. No crashes detected."
